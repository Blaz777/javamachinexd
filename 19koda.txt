class StavekIf {
     public static void main(String[] args){
          int prvi = 1, drugi = 2, rezultat;  
          boolean pogojIN; //tole je logièna spremenljivka, ki je pogoj, vkljuèen v stavek IF
 
          pogojIN = ((prvi == 1) && (drugi == 1)); //pogoju smo s primerjanjem in logièno operacijo priredili vrednost
          if (pogojIN) System.out.println(1. Tale stavek se izpiše le tedaj, ko je pogoj v stavku \"if\" izpolnjen!");

          pogojIN = ((prvi = 1) && (drugi == 2)); //pogoj smo nekoliko spremenili, tako, da je sedaj izpolnjen
          if (pogojIN) System.out.println("2. Tale stavek se izpiše le tedaj, ko je pogoj v stavku \"if\" izpolnjen!");
     }
}
